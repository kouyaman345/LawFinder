MATCH (l:Law)-[r:REFERENCES]->(t:Law)
RETURN l, r, t
LIMIT 100

// 最も参照される法令
MATCH (l:Law)<-[r:REFERENCES]-(from:Law)
RETURN l.id, l.title, count(r) as references
ORDER BY references DESC
LIMIT 10

// 参照タイプ別統計
MATCH ()-[r:REFERENCES]->()
RETURN r.type, count(r) as count
ORDER BY count DESC
